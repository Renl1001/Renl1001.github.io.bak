## [A - 一棵简单的线段树](http://qscoj.cn/#/contest/show/209)
### 题目大意
给定一个数组$a_1,a_2,...,a_n$初始的时候全部是0，然后有两个操作：1、单点替换；2、计算区间的和减去这个区间的最大值和最小值。

### 分析
简单的线段树模板题  
线段树功能：单点替换，区间求和，区间最大最小值

### 代码

```clike
#include <bits/stdc++.h>
using namespace std;
#define LL long long
#define INF 0x3f3f3f3f
#define lson l,m,rt<<1
#define rson m+1,r,rt<<1|1
#define ln rt<<1
#define rn rt<<1|1

const LL MOD = 1e9 + 7;
const int MAXN = 1e6 + 5;

LL sum[MAXN << 2];
LL MAX[MAXN << 2];
LL MIN[MAXN << 2];

void pushUp(int rt)
{
    sum[rt] = sum[ln] + sum[rn];
    MAX[rt] = max(MAX[ln], MAX[rn]);
    MIN[rt] = min(MIN[ln], MIN[rn]);
}

void build(int l, int r, int rt)
{
    if(l == r)
        return ;
    int m = (l + r) >> 1;
    build(lson);
    build(rson);
    pushUp(rt);
}

void update(int p, int c, int l, int r, int rt)
{
    if(l == r)
    {
        sum[rt] = c;
        MIN[rt] = c;
        MAX[rt] = c;
        return;
    }
    int m = (l + r) >> 1;
    if(p <= m) update(p, c, lson);
    else update(p, c, rson);
    pushUp(rt);
}

LL querySUM(int L, int R, int l, int r, int rt)
{
    if(L <= l && r <= R)
        return sum[rt];
    int m = (l + r) >> 1;
    LL ret = 0;
    if(L <= m) ret += querySUM(L, R, lson);
    if(m < R) ret += querySUM(L, R, rson);
    return ret;
}

LL queryMIN(int L, int R, int l, int r, int rt)
{
    if(L <= l && r <= R)
        return MIN[rt];
    int m = (l + r) >> 1;
    LL ret = INF;
    if(L <= m) ret = queryMIN(L, R, lson);
    if(m < R) ret = min(ret, queryMIN(L, R, rson));
    return ret;
}

LL queryMAX(int L, int R, int l, int r, int rt)
{
    if(L <= l && r <= R)
        return MAX[rt];
    int m = (l + r) >> 1;
    LL ret = -INF;
    if(L <= m) ret = queryMAX(L, R, lson);
    if(m < R) ret = max(ret, queryMAX(L, R, rson));
    return ret;
}

int main()
{
#ifndef Accepted
    // freopen("in.txt", "r", stdin);
#endif

    int n, m;
    cin >> n >> m;
    build(1, n, 1);
    while(m--)
    {
        int o, x, y;
        scanf("%d%d%d", &o, &x, &y);
        if(o == 0)
            update(x, y, 1, n, 1);
        else
        {
            LL sum = querySUM(x, y, 1, n, 1);
            LL minn = queryMIN(x, y, 1, n, 1);
            LL maxx = queryMAX(x, y, 1, n, 1);
            printf("%lld\n", sum - minn - maxx);
        }
    }
    return 0;
}
```

## [B - 一棵普通的线段树](http://qscoj.cn/#/contest/show/209)
### 题目大意
给定一个数组a1,a2,...,an初始的时候全部是0，然后有两个操作：1、选定一个区间，让该区间的每一个数都加v；2、计算区间和。

### 分析
简单的线段树模板题  
线段树功能：区间增减，区间求和

### 代码

```clike
#include <bits/stdc++.h>
using namespace std;
#define LL long long
#define INF 0x3f3f3f3f
#define lson l,m,rt<<1
#define rson m+1,r,rt<<1|1
#define ln rt<<1
#define rn rt<<1|1
const LL MOD = 1e9 + 7;
const int MAXN = 1e6 + 5;

LL sum[MAXN << 2];
LL lazy[MAXN << 2];

void pushUp(int rt)
{
    sum[rt] = sum[ln] + sum[rn];
}

void pushDown(int rt, int m)
{
    if(lazy[rt])
    {
        lazy[ln] += lazy[rt];
        lazy[rn] += lazy[rt];
        sum[ln] += lazy[rt] * (m - (m >> 1));
        sum[rn] += lazy[rt] * (m >> 1);
        lazy[rt] = 0;
    }
}

void build(int l, int r, int rt)
{
    lazy[rt] = 0;
    if(l == r)
    {
        sum[rt] = 0;
        return ;
    }
    int m = (l + r) >> 1;
    build(lson);
    build(rson);
    pushUp(rt);
}

void update(int L, int R, int v, int l, int r, int rt)
{
    if(L <= l && r <= R)
    {
        sum[rt] += 1ll * v * (r - l + 1);
        lazy[rt] += v;
        return ;
    }
    pushDown(rt, r - l + 1);
    int m = (l + r) >> 1;
    if(L <= m) update(L, R, v, lson);
    if(m < R) update(L, R, v, rson);
    pushUp(rt);
}

LL query(int L, int R, int l, int r, int rt)
{
    if(L <= l && r <= R)
        return sum[rt];
    pushDown(rt, r - l + 1);
    int m = (l + r) >> 1;
    LL ret = 0;
    if(L <= m) ret += query(L, R, lson);
    if(m < R) ret += query(L, R, rson);
    return ret;
}

int main()
{
#ifndef Accepted
    //freopen("in.txt", "r", stdin);
#endif

    int n, m;
    scanf("%d%d", &n, &m);
    build(1, n, 1);
    while(m--)
    {
        int o, l, r, v;
        scanf("%d%d%d%d", &o, &l, &r, &v);
        if(o == 0)
            update(l, r, v, 1, n, 1);
        else
            printf("%lld\n", query(l, r, 1, n, 1));
    }
    return 0;
}
```
